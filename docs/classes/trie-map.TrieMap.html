    Example

      .set(['src', 'classes'], 2)
      .set(['src', 'modules'], 2)
      .set(['docs'], 8);
    const directoryPathsWithTwoFiles = [];
    directoryFileCounts.find([], (value, prefix) => {
      if(value === 2) {
        directoryPathsWithTwoFiles.push(prefix);
      }
    });

      .set(['src', 'modules'], 6)
      .set(['docs'], 8);
    let totalFiles = 0;
    directoryFileCounts.forEach([], (value, prefix) => {

    // totalFiles (2 + 6 + 8) is now = 16
